 SELECT c.combo_id,
    c.attraction_1_id,
    c.attraction_2_id,
    c.combo_price,
    c.discount_percent,
    c.active,
    c.created_at,
    c.updated_at,
    c.name,
    c.attraction_ids,
    c.attraction_prices,
    c.total_price,
    c.image_url,
    c.create_slots,
    c.desktop_image_url,
    c.slug,
    c.meta_title,
    c.short_description,
    c.description,
    c.image_alt,
    c.desktop_image_alt,
    c.faq_items,
    c.head_schema,
    c.body_schema,
    c.footer_schema,
    c.stop_booking,
    COALESCE(json_agg(json_build_object('attraction_id', ca.attraction_id, 'title', a.title, 'price', ca.attraction_price, 'image_url', a.image_url, 'desktop_image_url', a.desktop_image_url, 'slug', a.slug, 'position_in_combo', ca.position_in_combo)) FILTER (WHERE ca.attraction_id IS NOT NULL), '[]'::json) AS attractions
   FROM combos c
     LEFT JOIN combo_attractions ca ON c.combo_id = ca.combo_id
     LEFT JOIN attractions a ON ca.attraction_id = a.attraction_id
  GROUP BY c.combo_id;
